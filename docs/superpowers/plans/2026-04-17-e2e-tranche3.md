## Task 1: Form-Validation Edge Cases

**File:** `tests/e2e/tests/validation/forms.spec.ts` (new)

Create:

```ts
import { test, expect } from '@playwright/test';

/**
 * Catches server-side 500 errors on malformed / malicious input. We're
 * not testing that the validation is *correct* — only that the server
 * doesn't crash, and the user stays on a sensible page.
 */
test.describe('Form Validation Edge Cases', () => {
  const SQL_INJECTION = "';DROP TABLE project;--";
  const LONG_STRING = 'x'.repeat(5000);
  const UNICODE = '日本語 🔥 Ω ∑ — مرحبا';

  async function assertNoServerError(page: import('@playwright/test').Page) {
    await expect(page.locator('body')).not.toContainText(/An internal server error occurred/i);
    await expect(page.locator('body')).not.toContainText(/SQLSTATE/);
    await expect(page.locator('body')).not.toContainText(/Exception/);
  }

  for (const payload of [SQL_INJECTION, LONG_STRING, UNICODE]) {
    const label = payload === LONG_STRING ? 'long-string' : payload === UNICODE ? 'unicode' : 'sql-injection';

    test(`project create survives ${label}`, async ({ page }) => {
      await page.goto('/project/create');
      await page.locator('input[name*="name"]').first().fill(payload);
      await page.locator('#form-project button[type="submit"], form button[type="submit"]').first().click();
      await assertNoServerError(page);
    });

    test(`credential create survives ${label}`, async ({ page }) => {
      await page.goto('/credential/create');
      const nameField = page.locator('input[name*="name"]').first();
      if (!(await nameField.isVisible({ timeout: 2_000 }).catch(() => false))) {
        test.skip(true, 'No credential create form available');
        return;
      }
      await nameField.fill(payload);
      await page.locator('form button[type="submit"]').first().click();
      await assertNoServerError(page);
    });
  }
});
```

**Notes:** The `form button[type="submit"]` fallback is used where the form has no discoverable id. The risk (logout-form match) was real on `/job-template/launch` (v2.2.3 incident). On `/project/create` and `/credential/create` the primary form is the FIRST visible form in the main content area; the layout's logout form renders below via the navbar partial — but to be safe, inspect `views/project/_form.php` and `views/credential/_form.php` to look for an explicit `id` and use that if present.

**Commit:** `test(e2e): guard project/credential create against malicious input`

## Task 6: Notification Send-Test Delivery (Mailhog)

**File:** `tests/e2e/tests/notification-templates/send-test.spec.ts` (new)

**Pre-check before writing:** read `tests/e2e/playwright.config.ts` to learn the base URL for Mailhog. In the compose stack Playwright can reach Mailhog at `http://mailhog:8025`. The API for messages is `GET /api/v2/messages`. There is also a DELETE endpoint to clear.

Create:

```ts
import { test, expect } from '@playwright/test';

/**
 * Proves that clicking "Send test" on a notification template actually
 * delivers an email to the configured SMTP (mailhog in the e2e stack).
 */
test.describe('Notification Template Send-Test', () => {
  const MAILHOG_API = 'http://mailhog:8025/api/v2';

  test('send-test email arrives at Mailhog', async ({ page, request }) => {
    // 1. Clear the Mailhog inbox so we see only this test's messages.
    await request.delete(`${MAILHOG_API}/messages`).catch(() => { /* may not exist on some Mailhog versions */ });

    // 2. Navigate to the seeded e2e-notification template.
    await page.goto('/notification-template/index');
    const row = page.locator('table.table tbody tr').filter({ hasText: 'e2e-notification' }).first();
    if (!(await row.isVisible({ timeout: 2_000 }).catch(() => false))) {
      test.skip(true, 'No e2e-notification template seeded');
      return;
    }
    const link = row.locator('a').first();
    const href = await link.getAttribute('href');
    if (!href) {
      test.skip(true, 'Row has no link');
      return;
    }
    await page.goto(href);

    // 3. Click the Send Test action. The form/button may use either
    //    a <form action*="send-test"> or a button with text.
    const sendForm = page.locator('form[action*="send-test"]').first();
    if (!(await sendForm.isVisible({ timeout: 2_000 }).catch(() => false))) {
      test.skip(true, 'Send-test form not visible on template view');
      return;
    }
    const submit = sendForm.locator('button[type="submit"]').first();
    await submit.click();

    // 4. Give the SMTP delivery a moment, then poll Mailhog for the message.
    let total = 0;
    for (let i = 0; i < 10; i++) {
      const res = await request.get(`${MAILHOG_API}/messages`);
      if (res.ok()) {
        const body = await res.json();
        total = typeof body.total === 'number' ? body.total : (body.count ?? 0);
        if (total > 0) break;
      }
      await new Promise((r) => setTimeout(r, 500));
    }
    expect(total).toBeGreaterThan(0);
  });
});
```

**Commit:** `test(e2e): notification send-test delivers to Mailhog`

test.describe('Audit Log Filters', () => {

    // Unfiltered: go to audit log index, record row count.
    await page.goto('/audit-log/index');

    // Filter by a specific action we know is seeded (user.login fires on

    const filteredRows = await page.locator('table.table tbody tr').count();

    expect(filteredRows).toBeGreaterThan(0);
    expect(filteredRows).toBeLessThanOrEqual(unfilteredRows);

    // Spot-check: every visible action cell should contain 'user.login'
    // (case-sensitive SQL LIKE, so substring match is expected).
    const actionCells = await page.locator('table.table tbody tr').allInnerTexts();
    for (const text of actionCells) {
      expect(text).toContain('user.login');
    }
  });

  test('filter by non-matching action yields empty table', async ({ page }) => {
    await page.goto('/audit-log/index?action=nonexistent.action.xyzzy');
    const count = await page.locator('table.table tbody tr').count();
    expect(count).toBe(0);
  });
});

## Task 9: Disabled User Cannot Log In

**File:** `tests/e2e/tests/users/disabled-login.spec.ts` (new)

Create:

```ts
import { test, expect } from '@playwright/test';

test.describe('Disabled User Login', () => {
  /**
   * Admin disables e2e-viewer via the toggle-status action, then logs out
   * and tries to log in as that viewer. The login must fail because
   * User::findByUsername filters by STATUS_ACTIVE.
   *
   * NOTE: this test mutates shared e2e state (toggles viewer status).
   * It re-enables the user at the end so later specs can still log in
   * as viewer. If you add tests that depend on the viewer's disabled
   * state, do them in the same test to keep the invariant.
   */
  test('disabled viewer cannot log in', async ({ page, request }) => {
    // Step 1: log in as admin via UI (simpler than working across
    //          project storageStates) and disable the viewer.
    await page.goto('/site/login');
    await page.locator('input[name*="username"]').fill('e2e-admin');
    await page.locator('input[name*="password"]').fill('E2eAdminPass1!');
    await page.locator('#login-form button[type="submit"], form button[type="submit"]').first().click();
    await page.waitForURL(/\/(site\/index)?$/, { timeout: 10_000 });

    // Step 2: find the viewer user's id.
    await page.goto('/user/index');
    const row = page.locator('table.table tbody tr').filter({ hasText: 'e2e-viewer' }).first();
    if (!(await row.isVisible({ timeout: 2_000 }).catch(() => false))) {
      test.skip(true, 'e2e-viewer not found in user list');
      return;
    }
    const link = row.locator('a').first();
    const href = await link.getAttribute('href');
    const match = href?.match(/id=(\d+)/);
    if (!match) {
      test.skip(true, 'Viewer link has no id');
      return;
    }
    const viewerId = match[1];

    // Step 3: toggle status to disabled via POST.
    const toggleResponse = await page.request.post(`/user/toggle-status?id=${viewerId}`, { data: {} });
    expect([200, 302]).toContain(toggleResponse.status());

    try {
      // Step 4: log out admin, then try to log in as viewer. Expect failure.
      await page.goto('/site/logout');
      // logout is a POST form in the nav — use direct POST.
      await page.request.post('/site/logout', { data: {} }).catch(() => {});

      await page.goto('/site/login');
      await page.locator('input[name*="username"]').fill('e2e-viewer');
      await page.locator('input[name*="password"]').fill('E2eViewerPass1!');
      await page.locator('#login-form button[type="submit"], form button[type="submit"]').first().click();
      // Expect to remain on /login with an error — NOT redirected to dashboard.
      await page.waitForTimeout(1_000);
      expect(page.url()).toMatch(/\/login/);
      await expect(page.locator('body')).toContainText(/invalid|incorrect|disabled|error/i);
    } finally {
      // Step 5: re-enable the viewer so subsequent tests still work.
      // We have to re-login as admin (session may be gone).
      await page.goto('/site/login');
      await page.locator('input[name*="username"]').fill('e2e-admin');
      await page.locator('input[name*="password"]').fill('E2eAdminPass1!');
      await page.locator('#login-form button[type="submit"], form button[type="submit"]').first().click();
      await page.waitForURL(/\/(site\/index)?$/, { timeout: 10_000 }).catch(() => {});
      await page.request.post(`/user/toggle-status?id=${viewerId}`, { data: {} });
    }
  });
});
```

**Commit:** `test(e2e): disabled viewer cannot log in (regression lock)`

**Important:** This spec runs with the default admin storageState but logs itself out and back in. It MUST re-enable the viewer in `finally` or other specs break. The toggle-status endpoint is POST-only and requires `user.update` permission — admin has it.
